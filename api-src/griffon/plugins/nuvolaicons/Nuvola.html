727     @Nonnull

736     @Nonnull
737     public static Nuvola findByDescription(@Nonnull String description) {
738         if (isBlank(description)) {
739             throw new IllegalArgumentException("Description " + description + " is not a valid Nuvola icon description");
740         }
741         if (!description.contains(":")) {
742             throw new IllegalArgumentException("Description " + description + " is not a valid Nuvola icon description");
743         }
744 
745         String[] parts = description.split(":");
746 
747         for (Nuvola nuvola : values()) {
748             if (nuvola.category.equalsIgnoreCase(parts[0]) && nuvola.description.equalsIgnoreCase(parts[1])) {
749                 return nuvola;
750             }
751         }
752         throw new IllegalArgumentException("Description " + description + " is not a a valid Nuvola icon description");
753     }
